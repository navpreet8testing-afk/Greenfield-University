   168→// Get current page filename for active state
   169→function getCurrentPage() {
   170→  const path = window.location.pathname;
   171→  const filename = path.split('/').pop() || 'index.html';
   172→  return filename === '/' ? 'index.html' : filename;
   173→}

   175→// Render mega menu HTML
   176→function renderMegaMenu(key) {
   177→  const m = megaMenuData[key];
   178→  if (!m) return '';
   179→  let colsHTML = '';
   180→  m.cols.forEach(col => {
   181→    let linksHTML = '';
   182→    col.links.forEach(link => {
   183→      linksHTML += `<li><a href="${link.href}">${link.label}</a></li>`;
   184→    });
   185→    colsHTML += `<div class="mega-col"><h4>${col.title}</h4><ul>${linksHTML}</ul></div>`;
   186→  });
   187→  return `<div class="mega-menu"><div class="mega-menu-inner">${colsHTML}</div></div>`;
   188→}

   403→// Initialize floating dock
   404→function initFloatingDock() {
   405→  const dock = document.getElementById('floatingDock');
   406→  const backToTop = document.getElementById('backToTop');
   407→  const dockLinks = dock.querySelectorAll('.dock-link');
   408→  const pill = document.getElementById('dockPill');
   409→  const total = dockLinks.length;
   410→
   411→  let lastScrollY = 0;
   412→  const onScroll = () => {
   413→    const scrollY = window.scrollY;
   414→    const velocity = Math.abs(scrollY - lastScrollY);
   415→
   416→    if (scrollY > window.innerHeight * 0.6) {
   417→      dock.classList.add('visible');
   418→    } else {
   419→      dock.classList.remove('visible');
   420→    }
   421→
   422→    if (velocity > 15) {
   423→      dock.style.transform = 'translateX(-50%) translateY(0) scale(0.95)';
   424→      dock.style.opacity = '0.85';
   425→    } else {
   426→      dock.style.transform = '';
   427→      dock.style.opacity = '';
   428→    }
   429→
   430→    if (scrollY > 400) {
   431→      backToTop.classList.add('visible');
   432→    } else {
   433→      backToTop.classList.remove('visible');
   434→    }
   435→
   436→    lastScrollY = scrollY;
   437→  };
   438→  window.addEventListener('scroll', onScroll, { passive: true });
   439→
   440→  backToTop.addEventListener('click', () => window.scrollTo({ top: 0, behavior: 'smooth' }));
   441→
   442→  // Pill hover
   443→  dockLinks.forEach((link, i) => {
   444→    link.addEventListener('mouseenter', () => {
   445→      const pct = 100 / total;
   446→      pill.style.left = `${i * pct}%`;
   447→      pill.style.width = `${pct}%`;
   448→    });
   449→  });
   450→  dock.addEventListener('mouseleave', () => {
   451→    pill.style.left = '-100%';
   452→  });
   453→}

   559→// Initialize Lenis smooth scroll
   560→function initLenis() {
   561→  if (typeof Lenis === 'undefined') return;
   562→  const lenis = new Lenis({
   563→    duration: 1.2,
   564→    easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
   565→    smoothWheel: true,
   566→  });
   567→
   568→  lenis.on('scroll', ScrollTrigger.update);
   569→
   570→  gsap.ticker.add((time) => {
   571→    lenis.raf(time * 1000);
   572→  });
   573→  gsap.ticker.lagSmoothing(0);
   574→
   575→  // Smooth scroll for anchor links
   576→  document.querySelectorAll('a[href^="#"]').forEach(a => {
   577→    a.addEventListener('click', (e) => {
   578→      const href = a.getAttribute('href');
   579→      if (href === '#') return;
   580→      e.preventDefault();
   581→      const target = document.querySelector(href);
   582→      if (target) lenis.scrollTo(target, { offset: -80 });
   583→    });
   584→  });
   585→}

   607→        header.classList.add('open');
